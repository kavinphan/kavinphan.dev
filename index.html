Arq

Service skeleton framework for Ruby using metaprogramming.

class Echo
  extend Arq::Action

  params  :ping
  returns :pong

  run do
    @pong = @ping
  end
end

result = Echo.call(ping: "hello world!")
result[:pong] # => "hello world!"
